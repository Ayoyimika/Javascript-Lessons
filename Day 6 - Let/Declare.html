// Variables defined with let cannot be redeclared. // You cannot accidentally redeclare a variable. let x = "John Doe"; let x = 0; // SyntaxError: 'x' has already been declared // With var you can: var x = "John Doe"; var x = 0; BLOCK Scope { let x = 2; } // x can NOT be used here { var x = 2; } // x CAN be used here
